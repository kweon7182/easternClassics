\begin{luacode*}
   parsing = require("lua.parsing")   
   binding.functions["characterWithText"] = function(self,char,nums,txt1,txt2)
      char,tone = parsing.diacriticsSeparation("〫","〬")(char)
      if nums == "" then nums = "0,0" end
      local ind = nums:find(",")
      local x,y = nums:sub(1,ind-1),nums:sub(ind+1,#nums)
      if tone == "〫" then x = -x end
      nums = tostring(-1/100*x)..","..tostring(-1/100*y)
      if txt2 == nil then
	 return self.functions["put"]
	 (self,string.format([[\typeA{%s}{%s}{%s}{%s}]],
           char,txt1,tone,nums))
      elseif #parsing.parsing(txt2) == 1 then
	 return self.functions["put"]
	 (self,string.format([[\typeC{%s}{%s}{%s}{%s}{%s}]],
           char,txt1,txt2,tone,nums))
      else
	 return self.functions["put"]
	 (self,string.format([[\typeB{%s}{%s}{%s}{%s}{%s}]],
           char,txt1,txt2,tone,nums))
      end
   end
\end{luacode*}
